import java.util.Scanner;

public class Caesar {

            public static String encrypt(String pt,int shift){

                        StringBuilder ct = new StringBuilder();

                        for(int i =0;i<pt.length();i++) {

                                    char ch = pt.charAt(i);

                                    if(Character.isLetter(ch)) {

                                                char b = Character.isUpperCase(ch)?'A':'a';

                                                ch= (char)(((ch-b+shift)%26)+b);

                                    }

                                    ct.append(ch);

                        }

                        return ct.toString();

            }

            public static String decrypt(String ct,int shift) {

                        return encrypt(ct,26-shift);

            }

            public static void main(String[] args) {

                        Scanner sc = new Scanner(System.in);

                        System.out.println("Enter msg: ");

                        String pt = sc.nextLine();

                        System.out.println("Enter shift: ");

                        int key = Integer.valueOf(sc.nextLine());

                        System.out.println("Plain: "+pt);

                        System.out.println("Shift: "+key);

                        String eMsg = encrypt(pt,key);

                        System.out.println("Enc msg: "+eMsg);

                        String dMsg = decrypt(eMsg,key);

                        System.out.println("Dec msg: "+dMsg);

            }

}

3. RailFence cipher

 

public class RailFence {

    public static String encrypt(String text, int railCount) {

        char[][] railMatrix = new char[railCount][text.length()];

        for (int i = 0; i < railCount; i++) {

            for (int j = 0; j < text.length(); j++) {

                railMatrix[i][j] = '\n';

            }

        }

 

        boolean directionDown = false;

        int row = 0, col = 0;

 

        for (int i = 0; i < text.length(); i++) {

            if (row == 0 || row == railCount - 1) {

                directionDown = !directionDown;

            }

 

            railMatrix[row][col++] = text.charAt(i);

 

            if (directionDown) {

                row++;

            } else {

                row--;

            }

        }

 

        StringBuilder result = new StringBuilder();

        for (int i = 0; i < railCount; i++) {

            for (int j = 0; j < text.length(); j++) {

                if (railMatrix[i][j] != '\n') {

                    result.append(railMatrix[i][j]);

                }

            }

        }

 

        return result.toString();

    }

 

    public static String decrypt(String text, int railCount) {

        char[][] railMatrix = new char[railCount][text.length()];

        for (int i = 0; i < railCount; i++) {

            for (int j = 0; j < text.length(); j++) {

                railMatrix[i][j] = '\n';

            }

        }

 

        boolean directionDown = false;

        int row = 0, col = 0;

 

        for (int i = 0; i < text.length(); i++) {

            if (row == 0) {

                directionDown = true;

            } else if (row == railCount - 1) {

                directionDown = false;

            }

            railMatrix[row][col++] = '*';

            if (directionDown) {

                row++;

            } else {

                row--;

            }

        }

 

        int index = 0;

        for (int i = 0; i < railCount; i++) {

            for (int j = 0; j < text.length(); j++) {

                if (railMatrix[i][j] == '*' && index < text.length()) {

                    railMatrix[i][j] = text.charAt(index++);

                }

            }

        }


        StringBuilder result = new StringBuilder();

        row = 0;

        col = 0;

        for (int i = 0; i < text.length(); i++) {

            if (row == 0) {

                directionDown = true;

            } else if (row == railCount - 1) {

                directionDown = false;

            }

 

            if (railMatrix[row][col] != '*') {

                result.append(railMatrix[row][col++]);

            }

 

            if (directionDown) {

                row++;

            } else {

                row--;

            }

        }

 

        return result.toString();

    }

 

    public static void main(String[] args) {

        String plaintext = "ATTACK AT ONCE!";

        int railCount = 3;

 

        String encryptedText = encrypt(plaintext, railCount);

        System.out.println("Encrypted: " + encryptedText);

 

        String decryptedText = decrypt(encryptedText, railCount);

        System.out.println("Decrypted: " + decryptedText);

    }

}

Python Algorithm:

 

def gcd(a, b):

    while b != 0:

        a, b = b, a % b

    return a

 

def mod_inverse(a, m):

    for x in range(1, m):

        if (a * x) % m == 1:

            return x

    return None

 

def rsa_key_generation(p, q):

    n = p * q

    phi = (p - 1) * (q - 1)

    e = 3 

    if gcd(e, phi) == 1:

        d = mod_inverse(e, phi)

        if d is not None:

            public_key = (e, n)

            private_key = (d, n)

            return public_key, private_key

    return None

 

def rsa_encrypt(plaintext, public_key):

    e, n = public_key

    encrypted_msg = pow(plaintext, e, n)

    return encrypted_msg

 

def rsa_decrypt(ciphertext, private_key):

    d, n = private_key

    decrypted_msg = pow(ciphertext, d, n)

    return decrypted_msg

public class RSA{

    public static void main(String[] args) {

        int p = 3;

        int q = 11;

        int n = p * q;

        int phi = (p - 1) * (q - 1);

        int e = 3;

        int d = modInverse(e, phi);

 

 

        int message = 9;

 

        int encrypted = modPow(message, e, n);

        System.out.println("Encrypted: " + encrypted);

 

        int decrypted = modPow(encrypted, d, n);

        System.out.println("Decrypted: " + decrypted);

    }

 

    public static int modPow(int base, int exponent, int modulus) {

        int result = 1;

        base = base % modulus;

        while (exponent > 0) {

            if (exponent % 2 == 1) {

                result = (result * base) % modulus;

            }

            exponent = exponent >> 1;

            base = (base * base) % modulus;

        }

        return result;

    }

 

    public static int modInverse(int a, int m) {

        int m0 = m;

        int y = 0, x = 1;

 

        if (m == 1) {

            return 0;

        }

 

        while (a > 1) {

            int q = a / m;

            int t = m;

 

            m = a % m;

            a = t;

            t = y;

 

            y = x - q * y;

            x = t;

        }

 

        if (x < 0) {

            x += m0;

        }

        return x;

    }

}

HMAC SHA-1 Example:

 

import javax.crypto.Mac;

import javax.crypto.spec.SecretKeySpec;

import java.security.InvalidKeyException;

import java.security.NoSuchAlgorithmException;

import java.util.Base64;

 

public class HMAC_SHA1 {

    public static void main(String[] args) {

        try {

            String message = "This is a secret message";

            String secretKey = "mySecretKey";

 

            byte[] secretKeyBytes = secretKey.getBytes();

            byte[] messageBytes = message.getBytes();

 

            Mac hmacSha1 = Mac.getInstance("HmacSHA1");

 

            SecretKeySpec secretKeySpec = new SecretKeySpec(secretKeyBytes, "HmacSHA1");

            hmacSha1.init(secretKeySpec);

 

            byte[] hmacBytes = hmacSha1.doFinal(messageBytes);

 

            String hmac = Base64.getEncoder().encodeToString(hmacBytes);

 

            System.out.println("Message: " + message);

            System.out.println("HMAC-SHA1: " + hmac);

 

        } catch (NoSuchAlgorithmException | InvalidKeyException e) {

            e.printStackTrace();

        }

    }

}
