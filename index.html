import numpy as np

class NeuralNetwork():
    def __init__(self):
        np.random.seed()
        # a numerical value that generates a new set or repeats pseudo-random numbers.
        #The seed is a starting point for a sequence of pseudorandom numbers.
        self.synaptic_weights = 2 * np.random.random((3, 1)) - 1
        #weights represent the strength of connections between neurons in different layers of a neural network.
        #(3, 1) indicates that you have 3 input neurons and 1 output neuron.
        #The process of adjusting these weights to minimize the difference between
        #predicted and actual outputs is known as training or learning.
    
    def sigmoid(self, x):
        return 1 / (1 + np.exp(-x))
    
    def sigmoid_derivative(self, x):
        return x * (1 - x)
    
    def train(self, training_inputs, training_outputs, training_iterations):
        for iteration in range(training_iterations):
            output = self.think(training_inputs)
            error = training_outputs - output
            adjustments = np.dot(training_inputs.T, error * self.sigmoid_derivative(output))
            self.synaptic_weights += adjustments
    
    def think(self, inputs):
        inputs = inputs.astype(float)
        output = self.sigmoid(np.dot(inputs, self.synaptic_weights))
        return output

if __name__ == "__main__":

import java.util.*;  
class CaesarCipher {
    public static String encrypt(String pt, int shift){        
        StringBuilder ct = new StringBuilder();
        for(int i=0; i<pt.length();i++){
            char ch = pt.charAt(i);
            if(Character.isLetter(ch)){
                char b = Character.isUpperCase(ch)?'A':'a';
                ch = (char)(((ch - b + shift)%26)+b);
            }
            ct.append(ch);
        }
        return ct.toString();
    }
    public static String decrypt(String ct, int shift){
        return encrypt(ct,26-shift);
    }

import java.util.*;
public class RailFenceCipher {
    public static String encrypt(String message, int rails) {
        StringBuilder[] railMatrix = new StringBuilder[rails];
        for (int i = 0; i < rails; i++) 
        railMatrix[i] = new StringBuilder();

        int rail = 0, step = 1;
        for (char c : message.toCharArray()) {
            railMatrix[rail].append(c);
            if (rail == 0) step = 1;
            else if (rail == rails - 1) step = -1;
            rail += step;
        }
        StringBuilder encryptedMessage = new StringBuilder();
        for (StringBuilder railString : railMatrix) encryptedMessage.append(railString);
        return encryptedMessage.toString();
    }

    public static String decrypt(String encryptedMessage, int rails) {
        int messageLength = encryptedMessage.length();
        StringBuilder[] railMatrix = new StringBuilder[rails];
        for (int i = 0; i < rails; i++) 
        railMatrix[i] = new StringBuilder();
        int rail = 0, step = 1;
        for (int i = 0; i < messageLength; i++) {
            railMatrix[rail].append(encryptedMessage.charAt(i));
            if (rail == 0) step = 1;
            else if (rail == rails - 1) step = -1;
            rail += step;
        }
        StringBuilder decryptedMessage = new StringBuilder();
        for (StringBuilder railString : railMatrix) decryptedMessage.append(railString);
        return decryptedMessage.toString();
    }

    public static void main(String[] args) {
        String message = "HELLOWORLD";
        int rails = 3;
        String encryptedMessage = encrypt(message, rails);
        System.out.println("Encrypted: " + encryptedMessage);
        String decryptedMessage = decrypt(encryptedMessage, rails);
        System.out.println("Decrypted: " + decryptedMessage);
    }
}
